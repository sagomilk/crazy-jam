参数

样式

 
var headroom = new Headroom(elem, {
  "tolerance": 5,
  "offset": 1,
  "classes": {
    "initial": "animated",
    "pinned": "bounceInDown",
    "unpinned": "bounceOutUp"
  }
});
headroom.init();

// to destroy
headroom.destroy();
$("header").headroom({
  "tolerance": 5,
  "offset": 1,
  "classes": {
    "initial": "animated",
    "pinned": "bounceInDown",
    "unpinned": "bounceOutUp"
  }
});

// to destroy
$("#header").headroom("destroy");
